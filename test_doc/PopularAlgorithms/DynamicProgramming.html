4032 def dp(text1, text2, i, j, memo)
4033   # base case for exiting. When either text index reached the full size, that means it finished iterating
4034   if i == text1.size || j == text2.size
4035     return 0
4036   end
4037 
4038   unless memo[i][j].nil?
