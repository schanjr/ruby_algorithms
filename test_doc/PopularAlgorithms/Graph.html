4430 def bellman_ford_find_cheapest_price(n, flights, src, dst, k)
4431   # dp represents the cost of traveling for a particular Edge. Set all edge costs to infinity
4432   # in the beginning to find the min cost later
4433   dp = Array.new(n, Float::INFINITY)
4434   # Set starting point cost to 0.
4435   dp[src] = 0
4436 
4437   # Use the Bellman-Ford algorithm
4438   # Let E = the number of edges (connection between two edges - the line)
4439   # Let V = the number of vertices (nodes)
4440   # Let S = the id of the starting node
4441   # Let D = the array of size V that tracks the best distance from S to each node.
4442   # Bellman ford algorithm states V iterations to find the true cost. (Count from zero to V-1, which basically means V). However below substitutes
4443   # V with k stops. K == 1 means take one stop during the flight. That means you can travel at least 2 nodes.
4444   (0..k).each do
4445     new_dp = dp.dup
4446     flights.each do |flight|
4447       new_dp[flight[1]] = [new_dp[flight[1]], dp[flight[0]] + flight[2]].min
4448     end
4449     dp = new_dp
4450   end
4451 
4452   # Return the min cost to reach the dest node with at most k stops
4453   dp[dst] != Float::INFINITY ? dp[dst] : -1
4454 end

4456 def dijkstra_find_cheapest_price(n, flights, src, dst, k)
4457   neighbors = Hash.new { |h, k| h[k] = [] }
4458   flights.each { |from, to, price| neighbors[from] << [price, to] }
4459   queue = neighbors[src].map { |price, to| [price, to, k] }.sort
4460   visited = { src => k }
4461   until queue.empty?
4462     price, city, remaining = queue.shift
4463     visited[city] = remaining
4464     return price if city == dst
4465     neighbors[city].each { |cost, to|
4466       next if visited[to] && visited[to] >= remaining
4467       next_price = price + cost
4468       index = queue.bsearch_index { |price, _, _| price > next_price } || queue.size
4469       queue.insert(index, [next_price, to, remaining - 1])
4470     } if remaining > 0
4471   end
4472   -1
4473 end
