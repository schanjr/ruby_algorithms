3476   queue = [s]
3477   # Initialize a set to store the strings that have already been processed
3478   visited = Set.new

3483   while !queue.empty? do
3484     # Get the next string from the queue
3485     str = queue.shift
3486     # Check if the string is valid
3487     if valid_string?(str)
3488       # If it is valid, add it to the results and set the found flag to true
3489       found = true
3490       results << str
3491     end
3492     # If we have already found a valid string, we can stop processing the queue
3493     next if found
3494     # Otherwise, we need to remove one character at a time and add the modified strings to the queue
3495     (0...str.length).each do |i|
3496       # Only consider removing characters that are parentheses
3497       next unless str[i] == '(' || str[i] == ')'
3498       # Remove the character at index i

3500       # Skip this string if we have already processed it
3501       next if visited.include?(modified_str)
3502       # Add the modified string to the queue and the visited set

3513 def valid_string?(str)
3514   # Initialize a counter to keep track of the number of open parentheses
3515   counter = 0
